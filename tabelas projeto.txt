CREATE TABLE acoes (
    id_acao INT primary key not null,
    item1_id INT,
    item2_id INT,
    descricaoComb text,
    nome_acao VARCHAR(50),

    descricao TEXT,

    FOREIGN KEY (item2_id) REFERENCES itens(id_item)

CREATE TABLE inventario (
    inventario_id int auto_increment primary key,
    item_id int not null,
    nome_item varchar(50),
    quantidade int not null,
    id_save int, 
    FOREIGN KEY (item_id) REFERENCES itens(id_item)
);


select * from saves
create table saves(
	id_save int not null primary key auto_increment,
	id_cena_atual int not null,
	FOREIGN KEY (id_cena_atual) REFERENCES cenas(id_cena)
);
