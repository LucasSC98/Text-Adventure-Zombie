CREATE TABLE cenas (
    id_cena int not null primary key,
    nome varchar(50) not null,
    id_local INT,
    descricao_historia TEXT,
    help varchar (!00)
);


CREATE TABLE itens (
    id_item int not null PRIMARY KEY,
    nome varchar(100) NOT NULL,
    descricao varchar (150),
    quantidade int not null
);

CREATE TABLE cena_itens (
    cena_id int,
    item_id int,
    posicao  varchar(100),
    PRIMARY KEY (cena_id, item_id),
    FOREIGN KEY (cena_id) REFERENCES cenas(id_cena),
    FOREIGN KEY (item_id) REFERENCES itens(id_item)
);



CREATE TABLE acoes (
    id_acao INT primary key not null,
    item1_id INT,
    item2_id INT,
    descricaoComb text,
    nome_acao VARCHAR(50),
    descricao_negativa TEXT,
    descricao TEXT,
    FOREIGN KEY (item1_id) REFERENCES itens(id_item),
    FOREIGN KEY (item2_id) REFERENCES itens(id_item)
);


CREATE TABLE inventario (
    id int not null primary key,
    item_nome varchar (50),
    item_id int,
    quantidade int,
    FOREIGN KEY (item_id) REFERENCES itens(id_item)  
);

create table saves (
	id_save int primary key auto_increment,
	id_cena_atual int not null,
	foreign key (id_cena_atual) references cenas(id_cena)
);
